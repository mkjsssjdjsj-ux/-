const DashboardView = ({ db, t }) => {

  const todaysOrders = db.orders.filter(o => new Date(o.date).setHours(0,0,0,0) === today);
  const totalRevenue = db.orders.reduce((sum, o) => sum + o.total, 0);
  const totalProfit = db.orders.reduce((sum, o) => sum + o.profit, 0);
  const todayRevenue = todaysOrders.reduce((sum, o) => sum + o.total, 0);
  const lowStockCount = db.products.filter(p => p.stock <= p.minStock).length;

    const data = [];
    for(let i=6; i>=0; i--) {
      const d = new Date(); d.setDate(d.getDate() - i);

        name: dayStr,
        Revenue: dayOrders.reduce((s, o) => s + o.total, 0),
        Profit: dayOrders.reduce((s, o) => s + o.profit, 0)

    return data;
  }, [db.orders]);

  return (
    <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }} className="space-y-6">

const POSView = ({ db, updateDb, lang, t, setPrintOrder }) => {
  const [cart, setCart] = useState([]);
  const [discount, setDiscount] = useState(0);
  const [selectedCustomer, setSelectedCustomer] = useState('');
  const [paymentMethod, setPaymentMethod] = useState('Cash');
  const [search, setSearch] = useState('');

  const filteredProducts = db.products.filter(p => 
    p.stock > 0 && 
    (p.nameEn.toLowerCase().includes(search.toLowerCase()) || 
     p.nameAr.includes(search) || 
     p.sku.includes(search))
  );

  const addToCart = (product) => {
    setCart(prev => {
      const existing = prev.find(item => item.id === product.id);
      if (existing) {
        if(existing.qty >= product.stock) return prev; // block over-stock
        return prev.map(item => item.id === product.id ? { ...item, qty: item.qty + 1 } : item);
      }
      return [...prev, { ...product, qty: 1, size: product.sizes[0] || '', color: product.colors[0] || '' }];
    });
  };

  const updateCartQty = (id, delta) => {
    setCart(prev => prev.map(item => {
      if(item.id === id) {
        const newQty = Math.max(1, Math.min(item.qty + delta, db.products.find(p=>p.id===id).stock));
        return {...item, qty: newQty};
      }
      return item;
    }));
  };

  const removeFromCart = (id) => setCart(prev => prev.filter(item => item.id !== id));

  const subtotal = cart.reduce((sum, item) => sum + (item.sellPrice * item.qty), 0);
  const taxAmount = (subtotal - discount) * (db.settings.taxRate / 100);
  const total = Math.max(0, subtotal - discount + taxAmount);
  const totalProfit = cart.reduce((sum, item) => sum + ((item.sellPrice - item.costPrice) * item.qty), 0) - discount;

  const handleCheckout = () => {
    if (cart.length === 0) return;

    // Deduct Stock
    const updatedProducts = db.products.map(p => {
      const cartItem = cart.find(c => c.id === p.id);
      if (cartItem) return { ...p, stock: p.stock - cartItem.qty };
      return p;
    });

    // Create Order
    const newOrder = {
      id: generateId('ORD'),
      orderNo: `INV-${Date.now().toString().slice(-6)}`,
      date: new Date().toISOString(),
      customerId: selectedCustomer,
      customerName: db.customers.find(c => c.id === selectedCustomer)?.name || 'Guest',
      items: cart,
      subtotal, discount, tax: taxAmount, total, profit: totalProfit,
      paymentMethod
    };

    updateDb({ products: updatedProducts, orders: [...db.orders, newOrder] });
    setPrintOrder(newOrder); // Triggers the print view and auto-print
    setCart([]);
    setDiscount(0);
    setSelectedCustomer('');
  };

  if (db.products.length === 0) return <EmptyState icon={ShoppingCart} title={t('emptyPOS')} description={t('emptyProducts')} actionLabel={t('add')} onAction={() => window.location.hash = 'products'}/>;

  return (
    <div className="h-[calc(100vh-8rem)] flex flex-col lg:flex-row gap-6">
      {/* Product Grid */}
      <div className="flex-1 flex flex-col h-full space-y-4">
        <Input placeholder={t('searchPlaceholder')} value={search} onChange={e => setSearch(e.target.value)} icon={Search} />
        <div className="flex-1 overflow-y-auto grid grid-cols-2 md:grid-cols-3 xl:grid-cols-4 gap-4 pr-2 custom-scrollbar pb-20 lg:pb-0">
          {filteredProducts.map(p => (
            <motion.div whileHover={{ scale: 1.02 }} whileTap={{ scale: 0.98 }} key={p.id} onClick={() => addToCart(p)}
              className="bg-white dark:bg-neutral-900 border border-neutral-200 dark:border-neutral-800 rounded-2xl p-4 cursor-pointer hover:border-amber-500 transition-colors flex flex-col"
            >
              <div className="h-24 bg-neutral-100 dark:bg-neutral-800 rounded-xl mb-3 flex items-center justify-center">
                <ShoppingBag size={32} className="text-neutral-400" />
              </div>
              <h4 className="font-semibold text-neutral-900 dark:text-white line-clamp-1">{lang === 'ar' ? p.nameAr : p.nameEn}</h4>
              <div className="mt-auto pt-2 flex justify-between items-end">
                <span className="text-amber-600 dark:text-amber-500 font-bold">{db.settings.currency}{p.sellPrice}</span>
                <span className="text-xs text-neutral-500">{p.stock} left</span>
              </div>
            </motion.div>
          ))}
        </div>
      </div>

      {/* Cart Sidebar */}

    if (prod) {

      setEditingId(null);

    if(confirm('Delete this product?')) updateDb({ products: db.products.filter(p => p.id !== id) });

};

const AdminLockScreen = ({ onUnlock, onCancel, t }) => {
  const [password, setPassword] = useState('');
  const [error, setError] = useState(false);

  const handleSubmit = (e) => {
    e.preventDefault();
    if (password === '2009') onUnlock();
    else { setError(true); setTimeout(() => setError(false), 2000); setPassword(''); }
  };

  return (
    <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} className="absolute inset-0 bg-neutral-100/90 dark:bg-neutral-950/90 backdrop-blur-xl z-50 flex items-center justify-center p-4">
      <Card className="max-w-md w-full p-8 text-center border-t-4 border-t-rose-500 shadow-2xl">
        <ShieldAlert size={48} className="mx-auto text-rose-500 mb-4" />
        <h2 className="text-2xl font-bold text-neutral-900 dark:text-white mb-2">{t('adminLockTitle')}</h2>
        <p className="text-neutral-500 mb-6">{t('adminLockSub')}</p>
        <form onSubmit={handleSubmit} className="space-y-4">
          <input type="password" value={password} onChange={e=>setPassword(e.target.value)} autoFocus
            className={`w-full bg-neutral-50 dark:bg-neutral-900 border rounded-xl py-4 text-center tracking-widest text-xl outline-none ${error ? 'border-rose-500' : 'border-neutral-200 dark:border-neutral-800 focus:border-rose-500'}`}
          />
          <div className="flex gap-3 mt-4">
             <Button variant="secondary" onClick={onCancel} className="flex-1">{t('cancel')}</Button>
             <Button type="submit" className="flex-1 !bg-rose-600 !text-white hover:shadow-rose-500/30">{t('unlockBtn')}</Button>
          </div>
        </form>
      </Card>
    </motion.div>
  );
};

// --- MAIN APP COMPONENT & DATA LAYER ---

export default function App() {
  const [auth, setAuth] = useState({ isAuthenticated: false, storeName: '' });
  const [adminUnlocked, setAdminUnlocked] = useState(false);
  const [currentView, setCurrentView] = useState('overview');
  const [lang, setLang] = useState(localStorage.getItem('aura_lang') || 'en');
  const [theme, setTheme] = useState(localStorage.getItem('aura_theme') || 'dark');
  const [db, setDb] = useState(() => JSON.parse(localStorage.getItem('aura_db')) || DEFAULT_DATA);
  const [printOrder, setPrintOrder] = useState(null);

  const t = (key) => TRANSLATIONS[lang][key] || key;

  // Persist DB
  useEffect(() => { localStorage.setItem('aura_db', JSON.stringify(db)); }, [db]);
  // Persist Preferences
  useEffect(() => { localStorage.setItem('aura_lang', lang); document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr'; }, [lang]);
  useEffect(() => { localStorage.setItem('aura_theme', theme); if (theme === 'dark') document.documentElement.classList.add('dark'); else document.documentElement.classList.remove('dark'); }, [theme]);

  // Handle Print Effect
  useEffect(() => {
    if (printOrder) {
      setTimeout(() => { window.print(); setPrintOrder(null); }, 500); // Slight delay to render
    }
  }, [printOrder]);

  // Global styles for print layout & scrollbars
  useEffect(() => {
    const style = document.createElement('style');
    style.innerHTML = `
      ::-webkit-scrollbar { width: 6px; height: 6px; }
      ::-webkit-scrollbar-track { background: transparent; }
      ::-webkit-scrollbar-thumb { background: #d4d4d8; border-radius: 10px; }
      .dark ::-webkit-scrollbar-thumb { background: #3f3f46; }
      body { background-color: ${theme === 'dark' ? '#0a0a0a' : '#fafafa'}; transition: background-color 0.3s; }
      @media print {
        body * { visibility: hidden; }
        .print-only, .print-only * { visibility: visible; }
        .print-only { position: absolute; left: 0; top: 0; width: 100%; display: block !important; }
      }
    `;
    document.head.appendChild(style);
    return () => document.head.removeChild(style);
  }, [theme]);

  const updateDb = (updates) => setDb(prev => ({ ...prev, ...updates }));

  if (!auth.isAuthenticated) {
    return <LoginScreen onLogin={(name) => setAuth({ isAuthenticated: true, storeName: name })} lang={lang} toggleLang={() => setLang(l => l === 'en' ? 'ar' : 'en')} theme={theme} toggleTheme={() => setTheme(t => t === 'dark' ? 'light' : 'dark')} t={t} />;
  }

  return (
    <>
      <InvoicePrint order={printOrder} storeName={auth.storeName} settings={db.settings} t={t} />
      
      <div className="no-print flex h-screen bg-neutral-50 dark:bg-neutral-950 text-neutral-900 dark:text-neutral-100 font-sans transition-colors duration-300">
        <Sidebar currentView={currentView} setCurrentView={setCurrentView} storeName={auth.storeName} t={t} />
        
        <div className="flex-1 flex flex-col h-screen overflow-hidden relative">
          <header className="h-20 border-b border-neutral-200 dark:border-neutral-800 bg-white/80 dark:bg-neutral-950/80 backdrop-blur-md flex items-center justify-end px-8 z-40">
            <div className="flex items-center gap-3">
              <button onClick={() => setLang(l => l === 'en' ? 'ar' : 'en')} className="p-2 bg-neutral-100 dark:bg-neutral-900 rounded-xl"><Languages size={18} /></button>
              <button onClick={() => setTheme(th => th === 'dark' ? 'light' : 'dark')} className="p-2 bg-neutral-100 dark:bg-neutral-900 rounded-xl">{theme === 'dark' ? <Sun size={18} /> : <Moon size={18} />}</button>
              <div className="h-6 w-px bg-neutral-200 dark:bg-neutral-800 mx-2"></div>
              <button onClick={() => { setAuth({ isAuthenticated: false, storeName: '' }); setAdminUnlocked(false); setCurrentView('overview'); }} className="flex items-center gap-2 text-rose-500 font-medium px-2"><LogOut size={18}/> <span className="hidden sm:block">{t('signOut')}</span></button>
            </div>
          </header>
          
          <main className="flex-1 overflow-y-auto p-6 md:p-8 z-10 custom-scrollbar relative">
            <AnimatePresence mode="wait">
              {currentView === 'admin-vault' && !adminUnlocked && (
                <AdminLockScreen key="lock" onUnlock={() => setAdminUnlocked(true)} onCancel={() => setCurrentView('overview')} t={t} />
              )}
              <motion.div key={currentView + (adminUnlocked ? '-unl' : '')} initial={{ opacity: 0, x: 20 }} animate={{ opacity: 1, x: 0 }} exit={{ opacity: 0, x: -20 }} transition={{ duration: 0.2 }} className="h-full max-w-7xl mx-auto">
                {currentView === 'overview' && <DashboardView db={db} t={t} />}
                {currentView === 'products' && <ProductsView db={db} updateDb={updateDb} lang={lang} t={t} />}
                {currentView === 'categories' && <CategoriesView db={db} updateDb={updateDb} lang={lang} t={t} />}
                {currentView === 'pos' && <POSView db={db} updateDb={updateDb} lang={lang} t={t} setPrintOrder={setPrintOrder} />}
                {currentView === 'orders' && <OrdersView db={db} t={t} setPrintOrder={setPrintOrder} />}
                {currentView === 'settings' && <SettingsView db={db} updateDb={updateDb} t={t} />}
                {currentView === 'admin-vault' && adminUnlocked && <AdminVaultView db={db} updateDb={updateDb} t={t} />}
              </motion.div>
            </AnimatePresence>
          </main>
        </div>
      </div>
    </>
  );
}
